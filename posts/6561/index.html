181.超过经理收入的员工

自连接就能解决

select e1.name as Employee
from employee e1, employee e2
where e1.managerid = e2.id 
and e1.salary > e2.salary;

182. 查找重复的电子邮箱

两种方法 第二种更简单直接

183. 从不订购的客户

# 子查询
select c.name as Customers
from customers c
where c.id not in (
  select customerID
  from orders 
)

# 左连接
select c.name as Customers
from customers c left join orders o
on c.id = o.customerID
where o.customerID is null

196. 删除重复的电子邮箱

delete p1 from person p1 join person p2
on p1.email = p2.email
where p1.id > p2.id

197. 上升的温度

考点是时间的差分函数 datediff()

select w1.id 
from weather w1 cross join weather w2
where datediff(w1.recorddate, w2.recorddate) = 1
  and w1.temperature > w2.temperature

586. 订单最多的客户

临时表，但是没必要

select customer_number
from (
  select customer_number, count(*) as counting
  from orders
  group by customer_number
) as temp
order by temp.counting desc
limit 1


# 直接gourp by 然后按照根据count的结果降序排序 选第一个
select customer_number
from orders
group by customer_number
order by count(customer_number) desc
limit 1

610. 判断三角形

解法1 if函数

select
    x,
    y,
    z,
    if(x+y>z and x+z>y and y+z>x , 'Yes', 'No') as triangle
from triangle;

解法2 case when函数

select x, y, z , 
  case when x+y>z and x+z>y and y+z>x then 'Yes'
  else 'No'
  end as triangle
from triangle

619. 只出现一次的最大数字

临时表先查出来 计数为1的，然后选最大的

但没有必要临时表 

select max(num) as num
from (  
    select num
    from mynumbers
    group by num 
    having count(num) = 1
) as temp

直接降序排序 只显示第一个 跟586.订单最多的客户一样

select num
from mynumbers
group by num
having count(num) = 1
order by num desc
limit 1

# 如果根本不存在 count等于1的数 会啥都不返回，所以一开始就加个判断
select if(count(num)=1,num,NULL) as num
from mynumbers
group by num
order by count(num), num desc
limit 1

627. 变更性别

要求用update语法：
update tablename
set …
where … 

# 解法1 if函数  一般就两种分类就用if  如果多种分类用case when 
update salary
set sex = if(sex = 'm', 'f', 'm')

update salary
set sex = case when sex = 'm' then 'f'
                when sex = 'f' then 'm'
                else sex
                end;         

update salary
set sex = (case when sex = 'f' then 'm'
          else 'f'
          end)

1075. 项目员工 I

# 冗长，可优化
select 
    distinct project_id,
    round(sum(e.experience_years)/count(project_id),2) as average_years
from
    project as p join employee as e
on p.employee_id = e.employee_id
group by project_id

# 直接用average就行了
select project_id, round(avg(experience_years),2) as average_years 
from project join employee
on project.employee_id = employee.employee_id
group by project_id

1084. 销售分析III 好题！

编写解决方案，报告2019年春季才售出的产品。即仅在2019-01-01至2019-03-31（含）之间出售的商品。
只在1月到3月销售，在其他月份没有销售。

# 不用连接 直接子查询
select product_id, product_name
from product
where product_id in (
    select product_id
    from sales 
    -- where sale_date between '2019-01-01' and '2019-03-31'  -- 这句可有可无
)
and product_id not in (
    select product_id
    from sales 
    where sale_date < '2019-01-01' or sale_date > '2019-03-31'
)

另一个NB的解法：
其实题目要求“仅在2019-01-01至2019-03-31之间出售的商品”翻译过来就是“所有售出日期都在这个时间内”，也就是“在这个时间内售出的商品数量等于总商品数量”

select sales.product_id as product_id, product.product_name as product_name
from sales left join product on sales.product_id = product.product_id
group by product_id
having count(sale_date between '2019-01-01' and '2019-03-31' or null) = count(*)

这里值得注意的是count的条件用法。举个例子count(age > 20 or null)这个语句，里面or null必须加，否则就等价于count(*)了，要么就是写作sum(age > 20)也可以。

我猜测是因为age > 20返回的是0或者1，而count对于不管是0还是1，都是会计数一次的，只有Null不会被计数。所以这个age > 20 or null表达的是不大于20就转换为null，这样就不会被count计数

  1. 查询结果的质量和占比 立马就用到了这个知识点 count(age > 20 or null) 等价于 sum(age > 20) 等价于 count(if(age > 20, 1, 0))

1141.查询近30天活跃用户数

-- 关键在于题目里说近30天
select 
    activity_date as day,
    count(distinct user_id) as active_users
from activity
-- where activity_date between '2019-06-28' and '2019-07-27'  --30天不能手算，要用datediff()
where datediff('2019-07-27', activity_date)<30
    and activity_date <= '2019-07-27'
    -- 或者其实等价于
    -- and datediff('2019-07-27',activity_date)>=0
group by activity_date

1211. 查询结果的质量和占比

select
    query_name,
    round(avg(rating/position),2) as quality,
    round(sum(rating < 3)/count(rating)*100, 2) as poor_query_percentage
    -- round(count(rating<3 or null)/count(rating)*100, 2) as poor_query_percentage
    -- ROUND(SUM(IF(rating < 3, 1, 0)) * 100 / COUNT(*), 2) poor_query_percentage
from queries
group by query_name

注意 三种写法都是等价的， 用sum(rating < 3) 最简单

1251.平均售价

1327. 列出指定时间段内所有的下单产品

select 
    p.product_name,
    sum(unit) as unit
from products as p left join orders as o
on p.product_id = o.product_id
-- where o.order_date between '2020-02-01' and '2020-02-29'  --2020年是闰年 有29天
where o.order_date like '2020-02%'
group by product_name
having sum(unit) >= 100

1179. 重新格式化部门表!!!

长变宽，不会且懒得做，先mark一下答案

SELECT id, 
SUM(CASE WHEN month='Jan' THEN revenue END) AS Jan_Revenue,
SUM(CASE WHEN month='Feb' THEN revenue END) AS Feb_Revenue,
SUM(CASE WHEN month='Mar' THEN revenue END) AS Mar_Revenue,
SUM(CASE WHEN month='Apr' THEN revenue END) AS Apr_Revenue,
SUM(CASE WHEN month='May' THEN revenue END) AS May_Revenue,
SUM(CASE WHEN month='Jun' THEN revenue END) AS Jun_Revenue,
SUM(CASE WHEN month='Jul' THEN revenue END) AS Jul_Revenue,
SUM(CASE WHEN month='Aug' THEN revenue END) AS Aug_Revenue,
SUM(CASE WHEN month='Sep' THEN revenue END) AS Sep_Revenue,
SUM(CASE WHEN month='Oct' THEN revenue END) AS Oct_Revenue,
SUM(CASE WHEN month='Nov' THEN revenue END) AS Nov_Revenue,
SUM(CASE WHEN month='Dec' THEN revenue END) AS Dec_Revenue
FROM department
GROUP BY id
ORDER BY id;

1280. 学生们参加各科测试的次数

考察多表连接

select 
    s.student_id, 
    s.student_name, 
    sub.subject_name,
    coalesce(count(e.subject_name),0) as attended_exams 
from Students s cross join Subjects sub left join Examinations e
on s.student_id = e.student_id
and sub.subject_name = e.subject_name
group by s.student_id,sub.subject_name
order by s.student_id,sub.subject_name;

1484. 按日期分组销售产品

考察group_concat函数

select 
    sell_date,
    count(distinct product) as num_sold,
    group_concat(distinct product separator ',') products
from activities
group by sell_date

例子：

id name
1 Alice
2 Bob
3 Carol
4 David

SELECT GROUP_CONCAT(name ORDER BY id ASC SEPARATOR ‘; ‘)
AS ordered_names
FROM students;

返回一个按 id 升序排列的名字，以分号和空格分隔，如 “Alice; Bob; Carol; David”。

1517. 查找拥有有效邮箱的用户

考察正则表达式

1、前缀 名称是一个字符串，可以包含字母（大写或小写），数字，下划线 ‘_’ ，点 ‘.’ 和/或破折号 ‘-‘ 。前缀名称 必须 以字母开头。
2、域为 ‘@leetcode.com’ 

SELECT *
FROM Users
WHERE mail REGEXP '^[a-zA-Z][a-zA-Z0-9_.-]*\\@leetcode\\.com$';

-- 很奇怪 _.- 换个顺序会报错 也需要转义才行
select *
from users
where mail regexp '^[a-zA-Z][a-zA-Z0-9\\.\\-\\_]*@leetcode\\.com$'
